    -H "Content-Type: application/json" \

    const url = new URL("http://localhost/api/updates");

let headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
}

let body = {
    "user_id": 12,
    "max_updates": 9
}

fetch(url, {
    method: "GET",
    headers: headers,
    body: body
