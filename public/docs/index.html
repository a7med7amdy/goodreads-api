    updates

    Get user's updates from following users


    Requires authentication

    Example request:

    curl -X GET -G "http://localhost/api/updates" \
    -H "Content-Type: application/json" \
    -d '{"user_id":18,"max_updates":11}'

    const url = new URL("http://localhost/api/updates");

let headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
}

let body = {
    "user_id": 18,
    "max_updates": 11
}

fetch(url, {
    method: "GET",
    headers: headers,
    body: body
})

    curl -X GET -G "http://localhost/api/changeCountry" \
    -H "Content-Type: application/json" \
    -d '{"country":"5WaGCn7VDAWGC3Mj"}'

    const url = new URL("http://localhost/api/changeCountry");

let headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
}

let body = {
    "country": "5WaGCn7VDAWGC3Mj"
}

fetch(url, {
    method: "GET",
    headers: headers,
    body: body
})
    .then(response => response.json())
    .then(json => console.log(json));

    Example response (200):

    {
    "status": "true",
    "message": "You have changed your country"
}
